IMPLEMENTATION OF ALGORITHM:

publickey = raw_input("Enter the public key: ")
generator = raw_input("Enter the generator: ")
privatekeyA = raw_input("Enter the private key of A: ")
privatekeyB = raw_input("Enter the private key of B: ")
asent = (pow(int(generator),int(privatekeyA)))%int(publickey)
bsent = (pow(int(generator),int(privatekeyB)))%int(publickey)
print "A sent ",asent," to B"
print "B sent ",bsent," to A"
agenerated = (pow(int(bsent),int(privatekeyA)))%int(publickey)
bgenerated = (pow(int(asent),int(privatekeyB)))%int(publickey)
print "The message generated by A is: ",agenerated
print "The message generated by B is: ",bgenerated
